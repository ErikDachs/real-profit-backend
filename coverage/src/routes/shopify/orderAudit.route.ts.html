function parseOrderId(raw: any): { ok: true; id: string } | { ok: false; status: number; error: string } {
  const s = String(raw ?? "").trim();
  if (!/^\d+$/.test(s)) return { ok: false, status: 400, error: "orderId must be a numeric Shopify order id" };
  return { ok: true, id: s };
}
